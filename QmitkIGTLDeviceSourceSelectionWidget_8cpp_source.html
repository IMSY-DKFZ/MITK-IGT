62void QmitkIGTLDeviceSourceSelectionWidget::IGTLDeviceSourceSelected(us::ServiceReferenceU s)
63 {
64 if (!s) //nothing selected

66 //reset everything

74 this->m_CurrentIGTLDeviceSource =
