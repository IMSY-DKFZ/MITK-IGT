58 if ( ! csvStream.is_open() ) {MITK_ERROR("USNavigationLoggingBackend") << "File '" << filename << "' cannot be opened for logging."; return;}
59 for (std::size_t i = 0; i < m_allNavigationMessages.size(); i++)
